12#include <windows.h>



13



14#define rtw_win_mutex_create( mutexDW ) \



15    *(mutexDW) = CreateMutex(NULL, 0U, NULL); 



16



17#define rtw_win_mutex_wait( mutexDW ) \



18    WaitForSingleObject(mutexDW, INFINITE);



19



20#define rtw_win_mutex_release( mutexDW ) \



21    ReleaseMutex(mutexDW);



22



23#define rtw_win_mutex_close( mutexDW ) \



24    CloseHandle(mutexDW);



25



26#define rtw_win_sem_create( semaphoreDW, initVal ) \



27    if ((initVal) == 0) {*(semaphoreDW) =  CreateSemaphore(NULL, 0, 1, NULL);} \



28    else {*(semaphoreDW) = CreateSemaphore(NULL, (initVal), (initVal), NULL);} 



29



30#define rtw_win_sem_wait( semaphoreDW ) \



31    WaitForSingleObject(semaphoreDW, INFINITE);



32



33#define rtw_win_sem_release( semaphoreDW ) \



34    ReleaseSemaphore(semaphoreDW, 1, NULL);



35



36#define rtw_win_sem_close( semaphoreDW ) \



37    CloseHandle(semaphoreDW);
